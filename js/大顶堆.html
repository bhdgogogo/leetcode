var halveArray = function (nums) {
  // ä¼˜å…ˆé˜Ÿåˆ— â€”â€” PriorityQueue

  let sum = nums.reduce((pre, cur) => {
    return cur + pre;
  }, 0);
  let half = sum / 2;
  let heap = new MyHeap(nums, (a, b) => a > b);

  let cnt = 0;
  while (sum > half) {
    let now = heap.pop();
    now /= 2;
    cnt++;
    sum -= now;
    heap.push(now);
  }
  return cnt;
};
